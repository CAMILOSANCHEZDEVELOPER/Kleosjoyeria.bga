            const enchantedLogic = (ctx, s) => {
                const w = ctx.canvas.width; const h = ctx.canvas.height;
                if (!s.p) {
                    s.p = [];
                    for (let i = 0; i < 60; i++) { // Increased particle count
                        s.p.push({ x: Math.random() * w, y: Math.random() * h, sz: Math.random() * 2 + 1, sx: (Math.random() - 0.5) * 0.4, sy: (Math.random() - 0.5) * 0.4, o: Math.random() });
                    }
                }
                ctx.clearRect(0, 0, w, h);
                ctx.globalCompositeOperation = 'lighter'; // Glow effect
                s.p.forEach(p => {
                    p.x += p.sx; p.y += p.sy; p.o += (Math.random() - 0.5) * 0.1;
                    if (p.o < 0) p.o = 0; if (p.o > 1) p.o = 1;
                    if (p.x > w || p.x < 0) p.sx *= -1;
                    if (p.y > h || p.y < 0) p.sy *= -1;
                    const gradient = ctx.createRadialGradient(p.x, p.y, 0, p.x, p.y, p.sz * 2.5);
                    gradient.addColorStop(0, `rgba(220, 180, 80, ${p.o * 0.9})`);
                    gradient.addColorStop(1, 'rgba(212, 175, 55, 0)');
                    ctx.fillStyle = gradient;
                    ctx.beginPath(); ctx.arc(p.x, p.y, p.sz * 2.5, 0, Math.PI * 2); ctx.fill();
                });
            };

            function setLanguage(lang) {
                document.documentElement.lang = lang;
                document.querySelectorAll('[data-lang]').forEach(el => {
                    const key = el.getAttribute('data-lang');
                    if (translations[lang] && translations[lang][key]) {
                        el.textContent = translations[lang][key];
                    }
                });
                document.querySelectorAll('[data-lang-placeholder]').forEach(el => {
                    const key = el.getAttribute('data-lang-placeholder');
                     if (translations[lang] && translations[lang][key]) {
                        el.placeholder = translations[lang][key];
                    }
                });
                // Update language display
                document.getElementById('current-lang').textContent = lang.toUpperCase();
            }

            function handleSendMessage() {
                const userMessage = chatInput.value.trim();
                if (userMessage === '') return;
                appendMessage(userMessage, 'user');
                chatInput.value = '';
                setTimeout(() => {
                    const currentLang = document.documentElement.lang || 'es';
                    const aiReply = getAiResponse(userMessage, currentLang);
                    appendMessage(aiReply, 'ai');
                }, 1000);
            }

            chatInput.addEventListener('keypress', (e) => e.key === 'Enter' && handleSendMessage());

            function appendMessage(message, sender) {
                const messageDiv = document.createElement('div');
                const messageP = document.createElement('p');
                messageP.textContent = message;

                if (sender === 'user') {
                    messageDiv.classList.add('text-right', 'mb-3');
                    messageP.classList.add('bg-gold', 'text-black', 'p-2', 'rounded-lg', 'inline-block');
                } else {
                    messageDiv.classList.add('mb-3');
                    messageP.classList.add('bg-gray-200', 'dark:bg-gray-700', 'text-gray-800', 'dark:text-gray-200', 'p-2', 'rounded-lg', 'inline-block');
                }
                messageDiv.appendChild(messageP);
                chatBody.appendChild(messageDiv);
                chatBody.scrollTop = chatBody.scrollHeight;
            }

            function getAiResponse(message, lang) {
                const lowerCaseMessage = message.toLowerCase();
                const responses = aiResponses[lang];
                for (const key in responses) {
                    if (lowerCaseMessage.includes(key)) return responses[key];
                }
                return responses.default;
            }

            // --- Enchanted Fireflies Effect ---
            const canvas = document.getElementById('firefly-canvas');
            const ctx = canvas.getContext('2d');
            let state = {}; 

            const enchantedLogic = (ctx, s) => {
                const w = ctx.canvas.width; const h = ctx.canvas.height;
                if (!s.p) {
                    s.p = [];
                    for (let i = 0; i < 60; i++) { // Increased particle count
                        s.p.push({ x: Math.random() * w, y: Math.random() * h, sz: Math.random() * 2 + 1, sx: (Math.random() - 0.5) * 0.4, sy: (Math.random() - 0.5) * 0.4, o: Math.random() });
                    }
                }
                ctx.clearRect(0, 0, w, h);
                ctx.globalCompositeOperation = 'lighter'; // Glow effect
                s.p.forEach(p => {
                    p.x += p.sx; p.y += p.sy; p.o += (Math.random() - 0.5) * 0.1;
                    if (p.o < 0) p.o = 0; if (p.o > 1) p.o = 1;
                    if (p.x > w || p.x < 0) p.sx *= -1;
                    if (p.y > h || p.y < 0) p.sy *= -1;
                    const gradient = ctx.createRadialGradient(p.x, p.y, 0, p.x, p.y, p.sz * 2.5);
                    gradient.addColorStop(0, `rgba(220, 180, 80, ${p.o * 0.9})`);
                    gradient.addColorStop(1, 'rgba(212, 175, 55, 0)');
                    ctx.fillStyle = gradient;
                    ctx.beginPath(); ctx.arc(p.x, p.y, p.sz * 2.5, 0, Math.PI * 2); ctx.fill();
                });
            };

            function resizeCanvas() {
                canvas.width = window.innerWidth;
                canvas.height = window.innerHeight;
                state = {}; 
            }

            function animate() {
                enchantedLogic(ctx, state);
                requestAnimationFrame(animate);
            }

            window.addEventListener('resize', resizeCanvas);
            resizeCanvas();
            animate();

            setLanguage('es'); 
        });
